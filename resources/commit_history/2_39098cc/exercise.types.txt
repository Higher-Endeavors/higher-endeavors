import { z } from 'zod';
import { LoadUnit } from './index';

// Exercise source type
export type ExerciseSource = 'library' | 'user' | 'custom';

// Base exercise properties
export interface BaseExercise {
  id: string;
  name: string;
  pairing: string;
  sets: number;
  reps: number;
  load: number | string;
  tempo: string;
  rest: number;
  notes?: string;
  rpe?: number;
  rir?: number;
  loadUnit?: LoadUnit;
  source?: ExerciseSource;
  libraryId?: number;
}

// Exercise with regular sets
export interface RegularExercise extends BaseExercise {
  isVariedSets: false;
  isAdvancedSets: false;
  setDetails?: undefined;
}

// Sub-set type for advanced sets
export interface SubSet {
  reps: number;
  load: number | string;
  loadUnit?: LoadUnit;
  rest: number;
}

// Set detail type for varied/advanced sets
export interface SetDetail {
  setNumber: number;
  reps: number;
  load: number | string;
  loadUnit?: LoadUnit;
  tempo: string;
  rest: number;
  notes?: string;
  subSets?: SubSet[];
}

// Exercise with varied sets
export interface VariedExercise extends BaseExercise {
  isVariedSets: true;
  isAdvancedSets: boolean;
  setDetails: SetDetail[];
}

// Union type for all exercise types
export type Exercise = RegularExercise | VariedExercise;

// Type guard functions
export const isRegularExercise = (exercise: Exercise): exercise is RegularExercise => {
  return !exercise.isVariedSets;
};

export const isVariedExercise = (exercise: Exercise): exercise is VariedExercise => {
  return exercise.isVariedSets;
};

// Zod schema for validation
export const exerciseSchema = z.discriminatedUnion('isVariedSets', [
  z.object({
    id: z.string(),
    name: z.string(),
    pairing: z.string(),
    sets: z.number(),
    reps: z.number(),
    load: z.union([z.number(), z.string()]),
    tempo: z.string(),
    rest: z.number(),
    notes: z.string().optional(),
    rpe: z.number().optional(),
    rir: z.number().optional(),
    loadUnit: z.enum(['kg', 'lbs']).optional(),
    source: z.enum(['library', 'user', 'custom']).optional(),
    libraryId: z.number().optional(),
    isVariedSets: z.literal(false),
    isAdvancedSets: z.literal(false),
    setDetails: z.undefined()
  }),
  z.object({
    id: z.string(),
    name: z.string(),
    pairing: z.string(),
    sets: z.number(),
    reps: z.number(),
    load: z.union([z.number(), z.string()]),
    tempo: z.string(),
    rest: z.number(),
    notes: z.string().optional(),
    rpe: z.number().optional(),
    rir: z.number().optional(),
    loadUnit: z.enum(['kg', 'lbs']).optional(),
    source: z.enum(['library', 'user', 'custom']).optional(),
    libraryId: z.number().optional(),
    isVariedSets: z.literal(true),
    isAdvancedSets: z.boolean(),
    setDetails: z.array(z.object({
      setNumber: z.number(),
      reps: z.number(),
      load: z.union([z.number(), z.string()]),
      loadUnit: z.enum(['kg', 'lbs']).optional(),
      tempo: z.string(),
      rest: z.number(),
      notes: z.string().optional(),
      subSets: z.array(z.object({
        reps: z.number(),
        load: z.union([z.number(), z.string()]),
        loadUnit: z.enum(['kg', 'lbs']).optional(),
        rest: z.number()
      })).optional()
    }))
  })
]);

// Helper functions to create properly typed exercises
export const createRegularExercise = (base: Omit<BaseExercise, 'isVariedSets' | 'isAdvancedSets'>): RegularExercise => ({
  ...base,
  isVariedSets: false,
  isAdvancedSets: false
});

export const createVariedExercise = (base: Omit<BaseExercise, 'isVariedSets' | 'isAdvancedSets'>, setDetails: SetDetail[]): VariedExercise => ({
  ...base,
  isVariedSets: true,
  isAdvancedSets: false,
  setDetails
});
